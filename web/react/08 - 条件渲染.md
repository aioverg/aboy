#### React - 条件渲染

基本思想是跟据条件来返回不同的组件，进而渲染不同的组件。

- 使用 `if` 实现条件渲染

  ```react
  <!DOCTYPE >
  <html>
  	<head>
  		<meta charset="utf-8" />
  		<title>React</title>
  		<!--引入react-->
  		<script src="react.development.js"></script>
  		<!--引入react-dom-->
  		<script src="react-dom.development.js"></script>
  		<!--引入babel，用以识别转义JSX-->
  		<script src="babel.min.js"></script>
  	</head>
  	<body>
  		<div id="root">占位</div>
  	</body>
  	<!--转义JSX-->
  	<script type="text/jsx">
  	    const root = document.getElementById('root')
  		
  		function Fx(props){ //函数组件
  		    return (
  			    <button onClick={props.onClick}>Fx</button>
  			)
  		}
  		
  		function Gx(props){ //函数组件
  		    return (
  			    <button onClick={props.onClick}>Gx</button>
  			)
  		}
  		
  	    class Ex extends React.Component {
  		    constructor(props){
  			    super(props)
  				this.f = this.f.bind(this)
  				this.g = this.g.bind(this)
  				this.state = {bool: true} 
  			}
  			f(){//修改state
  			    this.setState({bool: false})
  			}
  			g(){//修改state
  			    this.setState({bool: true})
  			}
  		    render(){
  			    let temp
  				if(this.state.bool){//当bool为true是加载<Fx />
  				    temp = <Fx onClick={this.f} />
  				}else{//当bool为false是加载<Gx />
  				    temp = <Gx onClick={this.g} />
  				}
  				
  			    return (
  				    <div>
  					    {temp}
  					</div>
  				)
  			
  			}
  		}
  		ReactDOM.render(<Ex />, root)
      </script>
  </html>
  ```

- 使用与运算符 `&&`

  ```react
  <!DOCTYPE >
  <html>
  	<head>
  		<meta charset="utf-8" />
  		<title>React</title>
  		<!--引入react-->
  		<script src="react.development.js"></script>
  		<!--引入react-dom-->
  		<script src="react-dom.development.js"></script>
  		<!--引入babel，用以识别转义JSX-->
  		<script src="babel.min.js"></script>
  	</head>
  	<body>
  		<div id="root">占位</div>
  	</body>
  	<!--转义JSX-->
  	<script type="text/jsx">
  	    const root = document.getElementById('root')
  		
  		function Ex(props){
  		    return(
  			    <h2>{props.age > 10 && props.age}</h2>
  			)
  		}
  		
  		ReactDOM.render(<Ex age={20} />, root)
      </script>
  </html>
  ```

  注意在`JavaScript` 中，`true && expression` 会返回 `expression` ，`false && expression` 会返回 `false` ，所以可以使用 `&&` 条件渲染。

- 使用三元运算符

  ```react
  <!DOCTYPE >
  <html>
  	<head>
  		<meta charset="utf-8" />
  		<title>React</title>
  		<!--引入react-->
  		<script src="react.development.js"></script>
  		<!--引入react-dom-->
  		<script src="react-dom.development.js"></script>
  		<!--引入babel，用以识别转义JSX-->
  		<script src="babel.min.js"></script>
  	</head>
  	<body>
  		<div id="root">占位</div>
  	</body>
  	<!--转义JSX-->
  	<script type="text/jsx">
  	    const root = document.getElementById('root')
  		
  		function Fx(props){ //定于函数组件
  		    return (<h2>{props.name}</h2>)
  		}
  		function Gx(props){ //定义函数组件
  		    return (<h2>{props.name}</h2>)
  		}
  		
  		class Ex extends React.Component {
  		    constructor(props){
  			    super(props)
  				this.state = {bool: true} //定义state
  				this.change = this.change.bind(this) //方法绑定this
  			}
  			change(){ //定义方法，用于修改this.state.bool
  			    this.setState({bool: !this.state.bool})
  			}
  			render(){
  			    return (
  			        <div>
                          //使用三元运算符渲染组件
  					    {this.state.bool ? <Fx name="hello" /> : <Gx name="world" />}
  				        <button onClick={this.change}>点击</button>
  				    </div>
  				)
  			}
  		
  		}
  		
  		ReactDOM.render(<Ex />, root)
      </script>
  </html>
  ```
